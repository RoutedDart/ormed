TUI skeleton template (Dart)

Use this as a quick starting point for a small interactive TUI.

---
import 'package:artisanal/tui.dart';

class AppModel implements Model {
  final int counter;
  const AppModel([this.counter = 0]);

  @override
  Cmd? init() => null;

  @override
  (Model, Cmd?) update(Msg msg) {
    return switch (msg) {
      KeyMsg(key: Key(type: KeyType.up)) => (AppModel(counter + 1), null),
      KeyMsg(key: Key(type: KeyType.down)) => (AppModel(counter - 1), null),
      KeyMsg(key: Key(type: KeyType.runes, runes: [0x71])) => (this, Cmd.quit()),
      _ => (this, null),
    };
  }

  @override
  String view() {
    return 'Counter: $counter\n\nUse up/down, q to quit';
  }
}

Future<void> main() async {
  await runProgram(const AppModel());
}
